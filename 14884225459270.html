一道JS的问题

题目:三个状态用红（stop）,绿（pass）,黄（wait）表示，要求用JavaScript让三个状态轮流切换，每个状态的停留时间是2秒。
HTML

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <link rel="stylesheet" href="index.css" type="text/css">
    <script src="index.js"></script>
</head>
<body>
<ul id="traffic">
    <li><span></span></li>
    <li><span></span></li>
    <li><span></span></li>
</ul>

</body>
</html>

CSS

li {
    list-style: none;
}

span {
    display: inline-block;
    width: 50px;
    height: 50px;
    border-radius: 50%;
    background-color: gray;
}
#traffic.wait li:nth-child(1) span{
    background-color: yellow;
}

#traffic.stop li:nth-child(2) span {
    background-color: red;
}

#traffic.go li:nth-child(3) span{
    background-color: green;
}

看到这个题目第一时间想到的就是setTimeout()的嵌套

//版本一
window.onload = function () {
    const traffic = document.getElementById('traffic');
    (function reset() {
        traffic.className = 'wait';
        setTimeout(function () {
            traffic.className = 'stop';
            setTimeout(function () {
                traffic.className = 'go';
                setTimeout(reset,2000)
            },2000)
        },2000)
    })()

}

这样写，可以解决问题，但是耦合性太强了，陷入callback hell，改一下代码

// 版本二
window.onload = function () {
    const traffic = document.getElementById('traffic');
    var stateList = ['wait','stop','go'];
    var currentStateIndex = 0;
    setInterval(function () {
        var state = stateList[currentStateIndex];
        traffic.className = state;
        currentStateIndex = (currentStateIndex + 1) % stateList.length;
    },2000)
}

根据灯状态的不同，建立一个数组，(currentStateIndex + 1) % stateList.length实现数组能够循环，比较好了，但是没有封装性，
我们继续修改

// 版本三
window.onload = function () {
    const traffic = document.getElementById('traffic');

    function start(traffic, stateList) {
        var currentStateIndex = 0;
        setInterval(function () {
            var state = stateList[currentStateIndex];
            traffic.className = state;
            currentStateIndex = (currentStateIndex + 1) % stateList.length;
        }, 2000)
    }
    start(traffic,['wait','stop','go']);
}

写了一个start（）方法，暴露出作用对象，状态数组，在其他的地方也可以使用
下面我们从函数式编程考虑，事件驱动，

// 版本四
window.onload = function () {
    const traffic = document.getElementById('traffic');
    function poll(...fnList) {
        let stateIndex = 0;
        return function (...args) {
            let fn = fnList[stateIndex++%fnList.length];
            return fn.apply(this,args);
        }
    }
    function setState(state) {
        traffic.className = state;
    }
    let trafficStatePoll = poll(setState.bind(null,'wait'),setState.bind(null,'stop'),setState.bind(null,'go'));
    setInterval(trafficStatePoll,2000);
}

现在我们更改一下需求，间隔时间分别是1，2，3秒

//现在改需求,1,2,3秒间隔,方案五
window.onload = function () {
    const traffic = document.getElementById('traffic');
    function wait(time) {
        return new  Promise(resolve => setTimeout(resolve,time));
    }
    function setState(state) {
        traffic.className = state;
    }
    function reset() {
        Promise.resolve()
            .then(setState.bind(null,'wait'))
            .then(wait.bind(null,1000))
            .then(setState.bind(null,'stop'))
            .then(wait.bind(null,2000))
            .then(setState.bind(null,'go'))
            .then(wait.bind(null,3000))
    }
    reset();

}

这里我们使用promise解决问题，抽象出了wait方法。
